📘 Como Configurar
  1. Crie uma planilha no Google Sheets com as abas: Produtos, Vendas, Despesas, Estoque
  2. No menu: Extensões → Apps Script
  3. Cole o código abaixo e salve:
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(data.type);
    
    if (!sheet) return ContentService.createTextOutput('{"error":"Aba não encontrada"}');
    
    if (data.type === 'Vendas') {
      sheet.appendRow([data.data.id, data.data.data, data.data.produto, 
        data.data.qtd, data.data.preco, data.data.total]);
    } else if (data.type === 'Produtos') {
      sheet.appendRow([data.data.id, data.data.nome, data.data.custo, 
        data.data.margem, data.data.preco, data.data.estoqueMin]);
    } else if (data.type === 'Despesas') {
      sheet.appendRow([data.data.id, data.data.data, data.data.desc, data.data.valor]);
    } else if (data.type === 'Estoque') {
      sheet.appendRow([data.data.id, data.data.produto, data.data.tipo, 
        data.data.qtd, data.data.data]);
    }
    
    return ContentService.createTextOutput('{"success":true}');
  } catch(err) {
    return ContentService.createTextOutput('{"error":"' + err + '"}');
  }
}
